source = pd.read_csv(r"D:\00_BACKUP_BY_DRIVE\cv\erdinc.csv")


source['start'] = pd.to_datetime(source['start'],yearfirst=True,format="%Y-%m-%d")
source['end'] = pd.to_datetime(source['end'],yearfirst=True,format="%Y-%m-%d")
source['diff'] = source['end'] - source['start']
source= source.sort_values(by='start').reset_index(drop=True)
dates_start=source["start"]
xp= source["experience"]
position= source["position"]

# Choose some nice levels
levels = np.tile([-5, 5, -3, 3, -1, 1],
                 int(np.ceil(len(dates_start)/6)))[:len(dates_start)]

# Create figure and plot a stem plot with the date
fig, ax = plt.subplots(figsize=(16, 5), layout="constrained")
ax.set(title="Experience Timeline")

ax.vlines(dates_start, 0, levels, color="tab:green")  # The vertical stems.
ax.plot(dates_start, np.zeros_like(dates_start), "-o",
        color="r", markerfacecolor="w")  # Baseline and markers on it.


# annotate lines
for d, l, x in zip(dates_start, levels, xp):
    ax.annotate(x, xy=(d, l),
                xytext=(-3, np.sign(l)*3), textcoords="offset points",
                horizontalalignment="center",
                verticalalignment="bottom" if l > 0 else "top")

for d, l, p in zip(dates_start, levels, position):
    ax.annotate(p, xy=(d, l),
                xytext=(-3, np.sign(l)*13), textcoords="offset points",
                horizontalalignment="center",
                verticalalignment="top" if l < 0 else "bottom")
    
# format x-axis with 12-month intervals
ax.xaxis.set_major_locator(mdates.YearLocator(1,8,1))
ax.xaxis.set_major_formatter(mdates.DateFormatter("%Y"))
#ax.xaxis.set_minor_locator(mdates.MonthLocator(interval=6))
#ax.xaxis.set_minor_formatter(mdates.DateFormatter("%b"))

plt.setp(ax.get_xticklabels(which="both"), rotation=90, ha="center")

# remove y-axis and spines
ax.yaxis.set_visible(False)
#ax.spines[["left", "top", "right"]].set_visible(False)

ax.margins(y=0.1)
plt.show()
